    <script>
        // --- Firestore Setup Variables (Required for deployment environment) ---
        // Note: These variables are used for any future database features, but they won't interfere with the static HTML functionality here.
        const appId = typeof __app_id !== 'undefined' ? __app_id : 'default-app-id';
        const firebaseConfig = typeof _firebase_config !== 'undefined' ? JSON.parse(_firebase_config) : {};

        // --- Frontend JavaScript Logic ---
        document.addEventListener('DOMContentLoaded', () => {
            const addToCartButtons = document.querySelectorAll('.add-to-cart');
            const cartCountElement = document.getElementById('cart-count');
            const cartMessage = document.getElementById('cart-message');
            const cartMessageText = document.getElementById('cart-message-text');
            let cartItemCount = 0;

            // Function to display the cart message
            function showCartMessage(text) {
                cartMessageText.textContent = text;
                cartMessage.classList.remove('hidden', 'translate-y-full');
                cartMessage.classList.add('translate-y-0');

                // Hide the message after 3 seconds
                setTimeout(() => {
                    cartMessage.classList.remove('translate-y-0');
                    cartMessage.classList.add('translate-y-full');
                    setTimeout(() => {
                        cartMessage.classList.add('hidden');
                    }, 300); // Wait for transition to finish
                }, 3000);
            }

            // Add click listener to all 'Add to Cart' buttons
            addToCartButtons.forEach(button => {
                button.addEventListener('click', (event) => {
                    // Prevent form submission if it were inside a form
                    event.preventDefault(); 
                    
                    // Increment cart count
                    cartItemCount++;
                    cartCountElement.textContent = cartItemCount;

                    // Find the product name for the notification
                    const productCard = button.closest('.product-card');
                    const productName = productCard ? productCard.querySelector('h3').textContent : 'Item';
                    
                    // Display success message
                    showCartMessage(${productName} added to cart!);
                });
            });

            // Handle the cart button click (can show a mock cart summary)
            document.getElementById('cart-button').addEventListener('click', () => {
                const message = cartItemCount === 0 
                    ? "Your cart is currently empty." 
                    : You have ${cartItemCount} item(s) in your cart.;
                
                showCartMessage(message);
            });
        });
    </script>
